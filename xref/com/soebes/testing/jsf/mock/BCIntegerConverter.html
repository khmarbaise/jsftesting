35  		for (int i = 0; i < sb.length(); i++) {

37  			if ((c == '0' && ersteZahlUngleichNull == false) || c == ' '

48  		// Wenn Eingabe komplett leer, 0 an erster Stelle einfügen

54  		if (max != null) {
55  			try {
56  				maxLength = Integer.parseInt(max);
57  			} catch (Exception e) {
58  			}
59  		}
60  
61  		if (sb.length() > maxLength)
62  			return ERROR_WERT_ZU_GROSS;

64  		return new Integer(Integer.parseInt(sb.toString()));
